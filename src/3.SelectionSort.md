function solve(N,arr){
    for(var i=0;i<N;i++){
        min=i;
        for(var j=i+1;j<N;j++){
            if(arr[j]<arr[min]){
                min=j;
            }
        }
        swap(arr,min,i);
    }
    console.log(arr.join(" "));
}

function swap(arr,min,i){
    var temp=arr[min];
    arr[min]=arr[i];
    arr[i]=temp;
}
